    "created": "string (date-time)",
    "dept": "string",
    "id": "integer (int64)",

    "name": "string"
  },
  "doctor_id": "integer (int64)",
  "id": "integer (int64)",
  "patient": {
    "age": "integer (int32)",
    "created": "string (date-time)",
    "gender": "string",
    "id": "integer (int64)",

    "name": "string",
    "occupation": "string",
    "symptom_summary": "string"
  },
  "patient_id": "integer (int64)",
  "prescription": "string"
}

name: string

  "dept": "string",
  "id": "integer (int64)",
  "joining": "string (date-time)",
